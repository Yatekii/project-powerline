% x: 0, 1, 3, 4
\def\POSxUp{0,3}
\def\POSxDown{1,4}
\def\POSy{0,1,3}
\begin{circuitikz}
    \foreach \x in \POSxUp{
        \foreach \y in \POSy {
            \draw
            (\x,\y) to[empty photodiode] (\x,\y+1)
            ;
        }
        \draw (\x,2.5) node[rotate=90] {\ldots};
    }

    \draw(2,2.5) node {$\times 15$};

    \foreach \x in \POSxDown{
        \foreach \y in \POSy {
            \draw
            (\x,\y+1) to[empty photodiode] (\x,\y)
            ;
        }
        \draw (\x,2.5) node[rotate=90] {\ldots};
    }

    % Connecting Lines between Strings
    \draw (0,4) -- (1,4);
    \draw (1,0) -- (3,0);
    \draw (3,4) -- (4,4);

    \draw (0,0) -- (0,-1.5) node[ocirc] {~~IN};
    \draw (4,0) -- (4,-1.5) node[ocirc] {~~OUT};
\end{circuitikz}
