% x: 0, 1, 3, 4
\def\POSxUp{0,3}
\def\POSxDown{1,4}
\def\POSy{0,1,3,4}
\begin{circuitikz}
    \foreach \x in \POSxUp{
        \foreach \y in \POSy {
            \draw
            (\x,\y) to[empty photodiode] (\x,\y+1)
            ;
        }
        \draw (\x,2.5) node[rotate=90] {\ldots};
    }
    \foreach \x in \POSxDown{
        \foreach \y in \POSy {
            \draw
            (\x,\y+1) to[empty photodiode] (\x,\y)
            ;
        }
        \draw (\x,2.5) node[rotate=90] {\ldots};
    }

    % Connecting Lines between Strings
    \draw (0,5) -- (1,5);
    \draw (1,0) -- (3,0);
    \draw (3,5) -- (4,5);

    \draw (0,0) -- (0,-0.5) node[ocirc] {~~IN};
    \draw (4,0) -- (4,-0.5) node[ocirc] {~~OUT};
\end{circuitikz}
